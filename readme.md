# basic 
git --version
git config user.name
git status 

# staging 
git add index.html
git add --all

# commit 
git commit -m "init"
# staging all and commit in one go
git commit -a -m "again" 

# commit history
git log

# help
git commit -help
git help --all

# branch
git branch new_branch_name
# list all branches
git branch 

# checkout to another branch
git checkout new_branch_name
